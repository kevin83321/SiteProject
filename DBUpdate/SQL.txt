CREATE TABLE `TW_Stock`.`TWSE_HistoricalPrice` (
  `Date` date NOT NULL,
  `Ticker` varchar(10) NOT NULL,
  `Open` float NOT NULL,
  `High` float NOT NULL,
  `Low` float NOT NULL,
  `Close` float NOT NULL,
  `Volume` bigint(20) NOT NULL,
  PRIMARY KEY (`Date`,`Ticker`)) ENGINE=InnoDB DEFAULT CHARSET=utf8;
